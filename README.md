```bash
npm install --save growl_task_library
```

## Usage

```jsx
import React, { Component } from 'react'

import MyComponent from 'growl_task_library'
import 'growl_task_library/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
